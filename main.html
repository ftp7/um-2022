def get_word_stat(word_path, article_path):

    # implement KMP
    class KMP:
        def partial(self, pattern):
            """ Calculate partial match table: String -> [Int]"""
            ret = [0]

            for i in range(1, len(pattern)):
                j = ret[i - 1]
                while j > 0 and pattern[j] != pattern[i]:
                    j = ret[j - 1]
                ret.append(j + 1 if pattern[j] == pattern[i] else j)
            return ret

        def search(self, T, P):
            """
            KMP search main algorithm: String -> String -> [Int]
            Return all the matching position of pattern string P in T
            """
            partial, ret, j = self.partial(P), [], 0

            for i in range(len(T)):
                while j > 0 and T[i] != P[j]:
                    j = partial[j - 1]
                if T[i] == P[j]: j += 1
                if j == len(P):
                    ret.append(i - (j - 1))
                    j = partial[j - 1]

            return ret

    # get cleaned data
    import re
    def get_content_list(path):
        result = []
        with open(path) as f:
            result = re.sub(r'[^\w\s]', '', f.read().lower())  # no .split() since using KMP
        return result

    words_list = get_content_list(word_path).split()
    article_text = get_content_list(article_path)


    def searching_kmp(pattern):
        kmp = KMP()
        search_result = kmp.search(article_text, pattern)
        return len(search_result)  # return number of match found in article

    result = sum(map(searching_kmp, words_list))  # get total number found

    # print(f"Country:\t{article_path}\nText:\t{word_path}\nMatche(s) found:\t{result}\n".upper())
    return result

      • #### Central Limit Theorem:
        • we take samples of size $n$ from a population (that has a true population mean $\mu$ and standard deviation of $\sigma$ and calculate the sample mean $x$
        • given that $n$ is sufficiently large ($n >30$), the sampling distribution of the means will be normally distributed with:
        • mean $x$ approximately equal to the population mean $\mu$
        • the standard deviation of the sampling distribution $\frac{\sigma}{\sqrt{n}}$, where $n=$ sample size
          • If original population is normally distributed, the CLT applies even with a smaller sample size #### Standard Error:
        • The standard deviation of a sampling distribution is also known as the standard error of the estimate of the mean
        • In many instances, we cannot know the population standard deviation $=\frac{\sigma}{\sqrt{n}}$, so we estimate the standard error using the sample standard deviation:
        • $ \frac{\text{standard deviation of our sample}}{\sqrt{\text{sample size}}} $
        • Note:
          • as sample size increases, standard error will decrease
            • as the population standard deviation increases, so will the standard error

def find_optimum_route(stores={}, country="", ITER_LIMIT=1):
    from queue import PriorityQueue
    class Graph:
        def __init__(self, num_of_vertices):
            self.v = num_of_vertices
            self.edges = [[-1 for i in range(num_of_vertices)] for j in range(num_of_vertices)]
            self.visited = []
        def add_edge(self, u, v, weight):
            self.edges[u][v] = weight
            self.edges[v][u] = weight
    def dijkstra(graph, start_vertex):
        D = {v:float('inf') for v in range(graph.v)}
        D[start_vertex] = 0

        pq = PriorityQueue()
        pq.put((0, start_vertex))

        while not pq.empty():
            (dist, current_vertex) = pq.get()
            graph.visited.append(current_vertex)

            for neighbor in range(graph.v):
                if graph.edges[current_vertex][neighbor] != -1:
                    distance = graph.edges[current_vertex][neighbor]
                    if neighbor not in graph.visited:
                        old_cost = D[neighbor]
                        new_cost = D[current_vertex] + distance
                        if new_cost < old_cost:
                            pq.put((new_cost, neighbor))
                            D[neighbor] = new_cost
        return D

the factors¶

factors that influences the ranking¶

  • P: Probability of positive sentiment
  • D: Total distance covered by truck
  • G: Price of gas per liter in USD
  • T_c: Time since article created
  • T_u Time since article last updated ## score
  • the probability of a positive sentiment should be the most influential factor for the score, however, we have to include two other factors. We were already tracking distances covered by truck in each of the 5 countries, so we felt that it made sense to include the price per liter of gas in each country. We felt comfortable having those 3 inputs make up the score for a ranking. ## decay

  • the next step is to decide how the rankings to fall over time. We wanted our algorithm to have rankings decay substantially in roughly 3 days.

  • Another factor that was relevant, was that we would likely update the existing article at some point. We wanted the ranking algorithm to account for this by giving newly updated content a boost in ranking. Since we store the datetime for the last update, we use it to generate a value that would be subtracted from the decay caused by the creation datetime. ## the algorithm

  • $\text{score}=\frac{200P - \frac{DG}{20} - 0.75G + 66}{1+T_c^{1.8} - |T_c-T_u|^{1.2}}$
